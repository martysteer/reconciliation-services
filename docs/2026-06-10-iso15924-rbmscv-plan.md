# ISO 15924 + RBMS CV Services Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add two reconciliation datasets — ISO 15924 script codes and the RBMS Controlled Vocabulary — as isolang-pattern services served by the existing runtime.

**Architecture:** One self-contained service directory per dataset (Makefile + stdlib `build_db.py` + metadata.json + stub README), wired into the root Makefile's native `data` pipeline. Runtime code untouched; one image rebuild merges the new metadata. Design spec: `docs/2026-06-10-iso15924-rbmscv-design.md`.

**Tech Stack:** GNU Make, Python stdlib (sqlite3, json, csv-free), sqlite-utils FTS5, Datasette runtime (unchanged).

```python
#!/usr/bin/env python3
"""Build ISO 15924 script codes database from the Unicode registry file.

Source format (semicolon-delimited, '#' comments):
  Code;N°;English Name;Nom français;PVA;Unicode Version;Date
"""

import sqlite3
import sys


def main(db_path, source_file):
    conn = sqlite3.connect(db_path)
    c = conn.cursor()

    c.execute('''CREATE TABLE IF NOT EXISTS scripts (
        id TEXT PRIMARY KEY,
        name TEXT,
        code_num TEXT,
        name_french TEXT,
        pva TEXT,
        unicode_version TEXT,
        date TEXT,
        searchText TEXT
    )''')

    count = 0
    with open(source_file, 'r', encoding='utf-8-sig') as f:
        for line in f:
            line = line.strip()
            if not line or line.startswith('#'):
                continue
            parts = line.split(';')
            if len(parts) < 7:
                continue
            code, num, name, french, pva, unicode_version, date = parts[:7]
            search = ' '.join(p for p in (name, french, pva, code) if p)
            c.execute('''INSERT INTO scripts
                (id, name, code_num, name_french, pva, unicode_version, date, searchText)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?)''',
                (code, name, num, french, pva, unicode_version, date, search))
            count += 1

    conn.commit()
    conn.close()

    if count == 0:
        print('ERROR: no scripts parsed -- source file empty or format changed',
              file=sys.stderr)
        sys.exit(1)
    print(f'✓ Database ready: {count} scripts')


if __name__ == '__main__':
    if len(sys.argv) != 3:
        print('Usage: build_db.py <db_path> <iso15924.txt>')
        sys.exit(1)
    main(sys.argv[1], sys.argv[2])
```

# Makefile for ISO 15924 Script Code Reconciliation Service
#
# Datasette-based reconciliation endpoint for ISO 15924 script codes
# (Latn, Cyrl, Arab, ...), compatible with OpenRefine's W3C Reconciliation API.
#
# Usage:
#   make build    - Setup venv + download dataset + build database
#   make serve    - Start reconciliation server
#   make status   - Show data status
#   make clean    - Remove database
#   make clean-all - Remove everything including venv

SHELL := /bin/bash
.SHELLFLAGS := -eu -o pipefail -c
.DEFAULT_GOAL := help

# =============================================================================
# Configuration
# =============================================================================
VENV_DIR := .venv
DATA_DIR := data

SOURCE_URL := https://unicode.org/iso15924/iso15924.txt
SOURCE_FILE := $(DATA_DIR)/iso15924.txt
SQLITE_DB := $(DATA_DIR)/iso15924.db
METADATA_JSON := iso15924.metadata.json

PORT ?= 8001

PYTHON := $(VENV_DIR)/bin/python3
PIP := $(VENV_DIR)/bin/pip
DATASETTE := $(VENV_DIR)/bin/datasette
SQLITE_UTILS := $(VENV_DIR)/bin/sqlite-utils
VENV_DONE := $(VENV_DIR)/.done

# =============================================================================
# Main Targets
# =============================================================================
.PHONY: build
build: $(VENV_DONE) $(SQLITE_DB)
	@echo ""
	@echo "✓ ISO 15924 Reconciliation Service ready!"
	@echo "Start with: make serve"
	@echo "Endpoint:   http://127.0.0.1:$(PORT)/iso15924/scripts/-/reconcile"

.PHONY: serve
serve: $(VENV_DONE) $(SQLITE_DB) $(METADATA_JSON)
	@echo "ISO 15924 Script Code Reconciliation Service"
	@echo "Endpoint: http://127.0.0.1:$(PORT)/iso15924/scripts/-/reconcile"
	@echo "Press Ctrl+C to stop"
	@$(DATASETTE) $(SQLITE_DB) \
		--metadata $(METADATA_JSON) \
		--port $(PORT) \
		--host $(if $(PUBLIC),0.0.0.0,127.0.0.1)

.PHONY: status
status:
	@echo "Files:"
	@if [ -f "$(VENV_DONE)" ]; then echo "  ✓ venv"; else echo "  ✗ venv"; fi
	@if [ -f $(SOURCE_FILE) ]; then echo "  ✓ $(SOURCE_FILE)"; else echo "  ✗ $(SOURCE_FILE)"; fi
	@if [ -f $(SQLITE_DB) ]; then echo "  ✓ $(SQLITE_DB) ($$(du -h $(SQLITE_DB) | cut -f1))"; else echo "  ✗ $(SQLITE_DB)"; fi
	@if [ -f $(SQLITE_DB) ]; then \
		sqlite3 $(SQLITE_DB) "SELECT '  Scripts: ' || COUNT(*) FROM scripts;"; \
	fi

.PHONY: clean
clean:
	@rm -f $(SQLITE_DB)
	@echo "✓ Database removed"

.PHONY: clean-all
clean-all:
	@rm -rf $(DATA_DIR) $(VENV_DIR)
	@echo "✓ All files removed"

# =============================================================================
# Virtual Environment
# =============================================================================
$(VENV_DIR)/.done:
	@echo "Setting up Python virtual environment..."
	@python3 -m venv $(VENV_DIR)
	@$(PIP) install --upgrade pip -q
	@$(PIP) install datasette datasette-reconcile sqlite-utils -q
	@touch $@
	@echo "✓ Virtual environment ready"

# =============================================================================
# Data Download
# =============================================================================
$(DATA_DIR):
	@mkdir -p $(DATA_DIR)

$(SOURCE_FILE): | $(DATA_DIR)
	@echo "Downloading ISO 15924 from the Unicode Consortium..."
	@curl -sL -o $@ "$(SOURCE_URL)"
	@echo "✓ Downloaded: $$(wc -l < $@ | tr -d ' ') lines"

# =============================================================================
# Database Build
# =============================================================================
$(SQLITE_DB): $(VENV_DONE) $(SOURCE_FILE) build_db.py
	@echo "Building SQLite database..."
	@rm -f $(SQLITE_DB)
	@$(PYTHON) build_db.py $(SQLITE_DB) $(SOURCE_FILE)
	@echo "Creating FTS index..."
	@$(SQLITE_UTILS) enable-fts $(SQLITE_DB) scripts searchText name --fts5 --create-triggers

$(METADATA_JSON):
	@echo "ERROR: $(METADATA_JSON) not found! Restore from git."
	@exit 1

# =============================================================================
# Help
# =============================================================================
.PHONY: help
help:
	@echo "ISO 15924 Script Code Reconciliation Service"
	@echo ""
	@echo "Commands:"
	@echo "  make build     Setup venv + download + database"
	@echo "  make serve     Start server (port $(PORT))"
	@echo "  make status    Show status"
	@echo "  make clean     Remove database"
	@echo "  make clean-all Remove everything"
	@echo ""
	@echo "Endpoint: http://127.0.0.1:$(PORT)/iso15924/scripts/-/reconcile"
```

#!/usr/bin/env python3
"""Build RBMS Controlled Vocabulary database from the LOC scheme JSON-LD.

Source: https://id.loc.gov/vocabulary/rbmscv.json -- a flat JSON-LD array
containing the MADS scheme plus ~1476 term entries. Term entries have an
@id under the rbmscv/ prefix and a madsrdf authoritativeLabel (values are
sometimes duplicated in the source; first wins).
"""

import json
import sqlite3
import sys

TERM_PREFIX = 'http://id.loc.gov/vocabulary/rbmscv/'
LABEL_KEY = 'http://www.loc.gov/mads/rdf/v1#authoritativeLabel'


def main(db_path, source_file):
    with open(source_file, 'r', encoding='utf-8') as f:
        entries = json.load(f)

    conn = sqlite3.connect(db_path)
    c = conn.cursor()
    c.execute('''CREATE TABLE IF NOT EXISTS terms (
        id TEXT PRIMARY KEY,
        uri TEXT,
        name TEXT,
        searchText TEXT
    )''')

    count = 0
    for entry in entries:
        uri = entry.get('@id', '')
        if not uri.startswith(TERM_PREFIX):
            continue
        labels = [v.get('@value', '') for v in entry.get(LABEL_KEY, [])]
        labels = [label for label in labels if label]
        if not labels:
            continue
        name = labels[0]
        term_id = uri[len(TERM_PREFIX):]
        c.execute('''INSERT OR IGNORE INTO terms (id, uri, name, searchText)
                     VALUES (?, ?, ?, ?)''',
                  (term_id, uri, name, name))
        count += 1

    conn.commit()
    conn.close()

    if count == 0:
        print('ERROR: no terms parsed -- source file empty or format changed',
              file=sys.stderr)
        sys.exit(1)
    print(f'✓ Database ready: {count} terms')


if __name__ == '__main__':
    if len(sys.argv) != 3:
        print('Usage: build_db.py <db_path> <rbmscv.json>')
        sys.exit(1)
    main(sys.argv[1], sys.argv[2])
```
